    $(function() {
            $("#dequeue").click(function() {
                $("div").queue(function() {
                    $(this).animate({marginLeft: '200px'}, 1000);
                    $(this).animate({marginLeft: '0'}, 1000);
                    $(this).dequeue(); // call next function in queue
                });
            });
            $("#fadein").click(function() {
                $("div").fadeIn(1000);
            });
            $("#fadeout").click(function() {
                $("div").fadeOut(1000);
            });
            $("#fadetoggle").click(function() {
                $("div").fadeToggle(1000);
            });
        });
